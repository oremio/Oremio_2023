#include <ctime> //clock_t, clock()
template <typename M, typename NM>
void test_moveable(M c1, NM c2, long &value)
{
    char buf[10];

    // 測試 move
    cout << "\n\ntest, with moveable elements" << endl;
    typedef typename iterator_traits<typename M::iterator>::value_type V1type;
    clock_t timeStart = clock();
    for (long i = 0; i < value; ++i)
    {
        snprintf(buf, 10, "%d", rand());
        auto ite = c1.end();
        c1.insert(ite, V1type(buf));
    }
    cout << "construction, milli-seconds : " << (clock() - timeStart) << endl;
    cout << "size()= " << c1.size() << endl;
    output_static_data(*(c1.begin()));

    timeStart = clock();
    M c11(c1);
    cout << "copy, milli-seconds : " << (clock() - timeStart) << endl;

    timeStart = clock();
    M c12(std::move(c1));
    cout << "move copy, milli-seconds : " << (clock() - timeStart) << endl;

    timeStart = clock();
    c11.swap(c12);
    cout << "swap, milli-seconds : " << (clock() - timeStart) << endl;

    // 測試 non-moveable
    cout << "\n\ntest, with non-moveable elements" << endl;
    typedef typename iterator_traits<typename NM::iterator>::value_type V2type;
    timeStart = clock();
    for (long i = 0; i < value; ++i)
    {
        snprintf(buf, 10, "%d", rand());
        auto ite = c2.end();
        c2.insert(ite, V2type(buf));
    }

    cout << "construction, milli-seconds : " << (clock() - timeStart) << endl;
    cout << "size()= " << c2.size() << endl;
    output_static_data(*(c2.begin()));

    timeStart = clock();
    c21.swap(c22);
    cout << "swap, milli-seconds : " << (clock() - timeStart) << endl;
}

这段代码定义了一个名为 test_moveable 的模板函数，用于测试容器中包含可移动元素和不可移动元素的效果。

该函数有以下几个主要部分：

  1. 第一个部分是对可移动元素的测试。它首先循环插入随机生成的字符串元素到容器 c1 中，然后分别测试了复制构造、移动构造和交换操作的性能。测试过程中使用了 clock() 函数来记录时间，并输出各个操作的耗时。

  2. 第二个部分是对不可移动元素的测试，与可移动元素的测试类似，只是这次使用的是不可移动元素的容器 c2。同样地，它循环插入随机生成的字符串元素到容器中，然后测试了复制构造、移动构造和交换操作的性能。

这个函数使用了模板，并接受两个模板参数 M 和 NM，分别表示包含可移动元素的容器类型和不可移动元素的容器类型。在测试过程中，利用 iterator_traits 模板获取容器迭代器的值类型，并据此来确定插入元素的类型。

最后，这个函数会输出各个操作的耗时，并调用一个名为 output_static_data 的函数来输出 MyString 类中记录的静态数据成员的值。
